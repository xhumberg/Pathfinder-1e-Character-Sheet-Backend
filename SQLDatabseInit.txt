SELECT CharacterName FROM UserIDToEmail
INNER JOIN UserAccess ON UserIDToEmail.UserID = UserAccess.UserID
INNER JOIN PathfinderCharacter ON UserAccess.CharacterID = PathfinderCharacter.CharacterID
WHERE UserIDToEmail.UserEmail = 'xavierhumberg@gmail.com' OR UserAccess.UserID = -1;

SELECT ItemName, ItemCost, ItemDescription, Adjustments FROM PathfinderCharacter
INNER JOIN Equipment ON PathfinderCharacter.CharacterID = Equipment.CharacterID
INNER JOIN Items ON Items.ItemID = Equipment.ItemID
WHERE PathfinderCharacter.CharacterID = 0;

select CharacterName, ItemName, ItemSlot, ResourceName, ResourceRemaining, ResourceMax from Equipment
inner join PathfinderCharacter on PathfinderCharacter.CharacterID = Equipment.CharacterID
inner join Items on Items.ItemID = Equipment.ItemID
inner join TrackedResources on TrackedResources.ResourceID = Equipment.TrackedResourceID

select CharacterName, ResourceName, ResourceDescription, ResourceRemaining, ResourceMax from CharactersTrackedResources
inner join PathfinderCharacter on PathfinderCharacter.CharacterID = CharactersTrackedResources.CharacterID
inner join TrackedResources on TrackedResources.ResourceID = CharactersTrackedResources.TrackedResourceID

select SpellName, SpellsPrepped.SpellLevel from Spells
inner join SpellsPrepped on Spells.SpellID = SpellsPrepped.SpellID
where CharacterID = 0;

select StandardAdjustments.AdjustmentID, AdjustmentName, AdjustmentEffect from EnabledAdjustments
inner join StandardAdjustments on StandardAdjustments.AdjustmentID = EnabledAdjustments.AdjustmentID
where EnabledAdjustments.CharacterID = 0
